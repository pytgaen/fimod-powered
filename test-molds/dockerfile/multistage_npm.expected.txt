# ── Build stage ──────────────────────────────────────────────────
FROM node:22-slim AS builder

WORKDIR /app

COPY package.json package-lock.json* ./
RUN npm ci --omit=dev

# ── Runtime stage ────────────────────────────────────────────────
FROM node:22-slim

WORKDIR /app

COPY --from=builder /app/node_modules /app/node_modules

ENV NODE_ENV=production

COPY . .

EXPOSE 3000

CMD ["node","server.js"]
