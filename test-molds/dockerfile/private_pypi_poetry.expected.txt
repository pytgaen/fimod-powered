# ── Build stage ──────────────────────────────────────────────────
FROM python:3.12-slim AS builder

SHELL ["/bin/bash", "-o", "pipefail", "-c"]

ARG PYPI_USERNAME
ARG PYPI_PASSWORD
WORKDIR /app

ENV POETRY_HTTP_BASIC_PRIVATE_USERNAME="${PYPI_USERNAME}"
ENV POETRY_HTTP_BASIC_PRIVATE_PASSWORD="${PYPI_PASSWORD}"

RUN curl -sSL https://install.python-poetry.org | python -
ENV PATH="/root/.local/bin:$PATH"
RUN poetry config http-basic.private "${PYPI_USERNAME}" "${PYPI_PASSWORD}"
COPY pyproject.toml poetry.lock* ./
RUN poetry config virtualenvs.in-project true \
 && poetry install --only main --no-interaction --no-ansi

COPY . .

RUN poetry run python -c "import app; app.precompile()"

# ── Runtime stage ────────────────────────────────────────────────
FROM python:3.12-slim

SHELL ["/bin/bash", "-o", "pipefail", "-c"]

WORKDIR /app

COPY --from=builder /app/.venv /app/.venv
ENV PATH="/app/.venv/bin:$PATH"

COPY . .

CMD ["python","-m","app"]
